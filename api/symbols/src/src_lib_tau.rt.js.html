  48    * @private
  49    * @returns {array} List of URL strings

  51   function _headerScriptUrls() {
  52     var i, p, len, scripts,
  53         urls = [],
  54         head = document.getElementsByTagName('head')[0];
  55     if (head) {
  56       scripts = head.getElementsByTagName('script');
  57       for (i = 0, len = scripts.length; i < len; i++) {
  58         p = scripts[i].src.indexOf('?'); // Remove appended parameters
  59         urls.push(p === -1 ? scripts[i].src : scripts[i].src.substring(0, p));
  60       }
  61     }
  62     return urls;
  63   }

  75   /**
  76    * Returns a resolved URL for a relative (application) resource.
  77    * <p/>
  78    * Resource path must begin with a "/", "$shared", or be a full URL.
  79    * @private
  80    * @param {String} path App-relative path must be '/' prefixed
  81    * @param {String} appName Application name (optional)
  82    * @returns {String} Resolved path for the application name 
  83    */
  84   function _resolvePath(path, appName) {
  85     path = path.trim();
  86     var a = path.charAt(0) === '/';
  87     if ('$dashboard' == appName) {
  88       path = a ? path.substring(1): path;
  89     } else {
  90       if (appName && a) {
  91         path = _APPSPATH + appName.trim() + path;
  92       } else if (a) {
  93         path = '..' + path;
  94       } else if (/^\$shared\//.test(path)) {
  95         path = path.replace(/^\$shared\//, _SHAREDPATH);
  96       } else if (!/^(\w+:)?\/\/\/?/.test(path)) {
  97         tau.log.error('Relative resource must begin with "/": ' + path, 'tau.rt');
  98         return null;
  99       }
 100     }
 101     a = document.createElement('a');
 102     a.href = path;
 103     return a.href;
 104   }

 494           phase = capture ? 1 : 2;

 503     /**
 504      * Executes the callback functions for all listeners on the Event.
 505      * <p/>
 506      * Calling {@link Event.stopPropagation} at any time during event 
 507      * propagation will immediately call the EventPublisher's stoppedFn 
 508      * function; subsequently, it will interrupt, and stop, the rest of event 
 509      * propagation.
 510      * <p/>
 511      * Additionally, calling {@link Event.preventDefault} will prevent the 
 512      * EventPublisher's defaultFn from being called.
 513      * @param {Object} payload Data object to be delivered to the listener
 514      * @return {Number} Number of event listeners called 
 515      */
 516     notifyListeners: function (event, payload) {
 517       var i, len, ctxs, listeners,
 518           phase = event.eventPhase;
 519   
 520       if (phase == tau.rt.Event.AT_TARGET) { // retrieve both phase listeners
 521         listeners = this.getListeners(true);
 522         var add = this.getListeners(false);
 523         listeners = (listeners.length > 0) ? listeners.concat(add) : add;
 524         add = null;
 525       } else {
 526         listeners = this.getListeners(phase == tau.rt.Event.CAPTURING_PHASE); 
 527       }
 528       if ((phase < tau.rt.Event.BUBBLING_PHASE) && this._opts.alwaysDefault) {
 529         // antecedentFn is invoked on capture and target phase
 530         this._opts.antecedentFn.apply(this._source, arguments);
 531       }
 532       // Execute callback functions for all listeners
 533       for (i = 0, len = listeners.length; i < len; i++) {
 534         try {
 535           ctxs = listeners[i].$contexts;
 536           for (var j = 0, ctxlen = ctxs.length; j < ctxlen; j++) {
 537             listeners[i].$fn.apply(ctxs[j] || this._source, arguments);
 538           }
 539         } catch (ex) {
 540           // If graceful errors is set, log error and continue event callback
 541           if (this._opts.gracefulErrors) {
 542             tau.log.error('Error occured during event listener: ' + ex, this);
 543           } else {
 544             if (ex.stack) { // chrome
 545               tau.log(ex.stack);
 546             } else {
 547               throw ex;
 548             }
 549           }
 550         } finally {
 551           if (event._stopPropagation) {
 552             this._opts.stoppedFn.apply(this._source, arguments);
 553             break;
 554           }
 555         }
 556       }
 557       return i;
 558     },

 571     /**
 572      * Finds listener internal object using specified <code>callbackFn</code>
 573      * function. if not found, rturns <code>null</code>. The object which will
 574      * be returned comply with <code>{$fn: xxx, $contexts: []}</code>
 575      * @param callbackFn event listener function
 576      * @returns listener internal object
 577      */
 578     intern: function (callbackFn) {
 579       var listeners = this._listeners;
 580       for (var i = 0, len = listeners.length; i < len; i++) {
 581         if (listeners[i].$fn === callbackFn) {
 582           return listeners[i];
 583         }
 584       }
 585       return null;

 588     /**
 589      * Attaches an event listener callback function.
 590      * @param {Function} callbackFn Function to callback when the event occurs 
 591      * @param {Object} context CallbackFn's 'this' context
 592      * @param {Boolean} capture indicates event phase, if true then capturing
 593      * phase other than that bubbling up phase 
 594      * @returns {Boolean} true if the listener was correctly registered
 595      */
 596     addListener: function (callbackFn, context, capture) {
 597       if (!tau.isFunction(callbackFn)) {
 598         throw new TypeError('Specified event listener is not a function: '
 599             + callbackFn + this.currentStack());
 600       }
 601       if (context && !tau.isObject(context)) {
 602         throw new TypeError('Specified event context is not an Object type: '
 603             + context + this.currentStack());
 604       }
 605       var listener = this.intern(callbackFn);
 606       // 1(Capture), 2(Bubble), 3 (Both)
 607       callbackFn.capture |= (capture ? 1 : 2); // BIT-OR operation
 608       if (!listener) {
 609         listener = {'$fn': callbackFn, '$contexts': []};
 610         this._listeners.push(listener);
